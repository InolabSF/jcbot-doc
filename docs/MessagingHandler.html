def handle_by_messaging(messaging)
  # handle messaging json
    # message
  if messaging.include?('message')
    message = messaging['message']

    if message.include?('quick_reply') # quick_reply
      handle_postback(message['quick_reply'])
    elsif message.include?('text') # text
      handle_text(message['text'])
    elsif message.include?('attachments') # attachments
      handle_attachments(message['attachments'])
    end
  end
    # postback
  handle_postback(messaging['postback']) if messaging.include?('postback')

  # reply to sender
  post_messages
end
