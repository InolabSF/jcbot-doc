def self.create_if_needed(user_id, platform)
  sender = Sender.find_by_user_id(user_id)

  # create
  unless sender
    sender = Sender.new
    sender.user_id = user_id
    sender.platform = platform
    sender.save if sender.valid?
  end

  # get profile
  if sender
    Sidekiq::Client.enqueue(GetProfileTask, user_id) unless sender.first_name
  end
  sender
end
