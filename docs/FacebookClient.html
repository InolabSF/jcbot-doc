def get_profile
  uri_string = "https://graph.facebook.com/v2.9/#{@user_id}"
  connection = Faraday.new(:url => uri_string) do |faraday|
    faraday.request  :url_encoded
    faraday.response :logger
    faraday.adapter  Faraday.default_adapter
  end

  response = connection.get do |request|
    request.params['access_token'] = @access_token
    request.headers['Content-Type'] = 'application/json'
    request.options.timeout = 30
    request.options.open_timeout = 10
  end

  begin
    JSON.parse(response.body)
  rescue Exception => e
    {}
  end
end

def post_button(button)
  buttons = button['buttons']
  if buttons && buttons.count > MAX_NUMBER_OF_BUTTONS
    json = { 'text' => button['text'], 'quick_replies' => [] }
    buttons.each { |b| json['quick_replies'].push({ "content_type" => 'text', "title" => b['title'], "payload" => b['payload'] }) if b['payload'] }
    post_quick_replies(json)
  else
    # json
    message = {
      "attachment" => {
        "type" => 'template',
        "payload" => {
          "template_type" => 'button',
          "text" => "#{button['text']}",
          "buttons" => buttons,
        }
      }
    }
    # post
    post_message(message)
  end
end

def post_message(message)
  uri_string = 'https://graph.facebook.com/v2.9/me/messages'
  connection = Faraday.new(:url => uri_string) do |faraday|
    faraday.request  :url_encoded
    faraday.response :logger
    faraday.adapter  Faraday.default_adapter
  end

  json_to_post = { "recipient" => { "id" => @user_id }, "message" => message }
  response = connection.post do |request|
    request.params['access_token'] = @access_token
    request.headers['Content-Type'] = 'application/json'
    request.body = json_to_post.to_json
    request.options.timeout = 30
    request.options.open_timeout = 10
  end

  begin
    json_to_return = JSON.parse(response.body)

    # error handling
    #store_error_by_json({"platform" => 'facebook', "user_id" => @user_id, "json" => json_to_post}, json_to_return)

    json_to_return
  rescue Exception => e
    {}
  end
end

def post_thread_settings(menus)
  uri_string = "https://graph.facebook.com/v2.9/me/messenger_profile"
  connection = Faraday.new(:url => uri_string) do |faraday|
    faraday.request  :url_encoded
    faraday.response :logger
    faraday.adapter  Faraday.default_adapter
  end

  locales = {}
  menus.each do |menu|
    sender = Sender.find_by_id(menu.sender_id)
    next unless sender
    locale = sender.locale
    locales[:#{locale}"] = [] unless locales[:#{locale}"]
    action = nil
    if !(menu.payload.blank?)
      action = { :title => menu.text,  :type => 'postback', :payload => menu.payload }
    elsif !(menu.url.blank?)
      action = { :title => menu.text,  :type => 'web_url',  :url     => menu.url, :webview_height_ratio => 'full' }
    end
    next unless action
    locales[:#{locale}"].push(action)
  end

  json_to_post = { :persistent_menu => [] }
  locales.each do |locale, actions|
    persistent_menu = { :locale => (locale == :#{DEFAULT_LOCALE}") ? 'default' : locale, :call_to_actions => [] }
    persistent_menu[:call_to_actions] = actions
    json_to_post[:persistent_menu].push(persistent_menu)
  end

  response = connection.post do |request|
    request.params['access_token'] = @access_token
    request.headers['Content-Type'] = 'application/json'
    request.body = json_to_post.to_json
    request.options.timeout = 30
    request.options.open_timeout = 10
  end

  begin
    JSON.parse(response.body)
  rescue Exception => e
    {}
  end
end
