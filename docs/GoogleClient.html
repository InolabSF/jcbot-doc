def apiai_context_request(user_id, context_name)
  client = ApiAiRuby::Client.new(client_access_token: @apiai_key, api_session_id: user_id)
  context_request = client.create_contexts_request
  if context_name
    context_request.create(context_name)
  else
    context_request.delete()
  end
end

def get_spreadsheet
  session = GoogleDrive::Session.from_service_account_key('./lib/assets/third_party/google_client.json')
  worksheets = session.spreadsheet_by_key(@spreadsheet_key).worksheets

  worksheets_json = {}
  worksheets.each do |worksheet|
    column_names = []
    spreadsheets = []
    worksheet_name = worksheet[SPEADSHEET_MODEL_NAME_ROW, 1]
    next unless worksheet_name

    (1..worksheet.num_cols).each do |col|
      column_names.push(worksheet[SPEADSHEET_MODEL_COLUMN_NAME_ROW, col])
    end
    next if column_names.empty?

    start_row = SPEADSHEET_MODEL_COLUMN_NAME_ROW + 1
    (start_row..worksheet.num_rows).each do |row|
      spreadsheet = {}
      (1..worksheet.num_cols).each do |col|
        spreadsheet[column_names[col-1]] = worksheet[row, col]
      end
      spreadsheets.push(spreadsheet) unless spreadsheet.empty?
    end

    worksheets_json[worksheet_name] = spreadsheets
  end

  worksheets_json
end

def translate_request(text, lang_to)
  translate_lang = "#{lang_to}".downcase
  translate_lang = 'chinese_simplified' if translate_lang == 'chinese'
  translate_lang_code = nil
  EasyTranslate::LANGUAGES.each { |key, value| translate_lang_code = :#{key}" if value == translate_lang }
  return '' unless translate_lang_code
  EasyTranslate.translate(text, :to => translate_lang_code, :key => @api_key)
end
