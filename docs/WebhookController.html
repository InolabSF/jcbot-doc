def get_facebook
  verify_token = params['hub.verify_token']
  render text: 'invalid verify_token' and return unless verify_token == ENV['FB_VERIFY_TOKEN']
  challenge = params['hub.challenge']
  render text: 'invalid challenge' and return unless challenge

  render text: challenge
end

def post_facebook
  # response
  json = { :application_code => 200 }

  begin
    messaging = params['entry'][0]['messaging'][0]
    raise "messaging is needed" unless messaging

    user_id = messaging['sender']['id']
    raise "facebook user id is needed" unless user_id

  rescue => e
    json = { :application_code => 400, :message => e.message}
    render :json => json, :status => 400
    return
  end

  Sender.create_if_needed(user_id, Sender::FACEBOOK)
  messaging_handler = MessagingHandler.new(user_id)
  messaging_handler.handle_by_messaging(messaging)

  render json: json, status: :ok
end

def post_line
  line_client ||= Line::Bot::Client.new { |config|
    config.channel_secret = ENV["LINE_CHANNEL_SECRET"]
    config.channel_token = ENV["LINE_CHANNEL_ACCESS_TOKEN"]
  }

  json_converter = JsonConverter.new(Sender::LINE)
  facebook_json = json_converter.incoming_json(params)

  begin
    messaging = facebook_json['entry'][0]['messaging'][0]
    raise "messaging is needed" unless messaging

    user_id = messaging['sender']['id']
    raise "line user id is needed" unless user_id

  rescue => e
    json = { :application_code => 400, :message => e.message}
    render :json => json, :status => 400
    return
  end

  Sender.create_if_needed(user_id, Sender::LINE)
  messaging_handler = MessagingHandler.new(user_id)
  messaging_handler.handle_by_messaging(messaging)

  render :nothing => true, status: :ok
end
