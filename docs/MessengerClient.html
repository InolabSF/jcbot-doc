def initialize(sender)
  @sender = sender
  case @sender.platform
  when Sender::FACEBOOK
    @client = FacebookClient.new(@sender.user_id)
  when Sender::LINE
    @client = LineClient.new(@sender.user_id)
  else
    @client = nil
  end
end
