def perform
  @performed = true

  # import spreadsheets
  google_client = GoogleClient.new
  worksheets = google_client.get_spreadsheet
  worksheets.each do |model_name, spreadsheets|
    begin
      clazz = model_name.camelize.constantize
    rescue Exception => e
      next
    end
    next unless clazz

    spreadsheets.each do |spreadsheet|
      instance = clazz.find_or_create_by(id: spreadsheet["id"])
      spreadsheet.each { |key, value| instance[key] = value }
      instance.save if instance.valid?
    end
  end

  # post facebook get_started, persistent_menu
  menus = Menu.all
  facebook_client = FacebookClient.new(nil)
  facebook_client.post_get_started
  facebook_client.post_thread_settings(menus)
end
